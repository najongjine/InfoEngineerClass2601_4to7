
#include <stdio.h>

/* java class의 아버지.
class 보다 성능은 많이 떨어짐
상속 X method 같이 포함 X */
struct S {
	int x;
	short y;
};
/*
이거 자료결합도.
struct 을 printf 로 찍어보면 주소 안나옴. 쌩 값나옴
그러면 매개변수에 값 복붙해줄때, 주소 복붙이 아니라
쌩 값 복붙임
*/
void change(struct S s) {
	s.x += 10;
	s.y += 10;
}
int main()
{
	int      a =   1;
	int ar[] = { 1,2 };
	/* s 는 주소 저장하는게 아님!!! 그냥 쌩 원본 데이터 저장함 */
	struct S s = { 2,4 };
	printf("s: %d \n ", s); // 2
	printf("s.x: %d \n ", s.x); // 2
	printf("s.y: %d \n ", s.y); // 4 

	struct S s2 = s;
	s2.x = 99;
	printf("s.x: %d \n ", s.x); 
	printf("s.y: %d \n ", s.y);  
	printf("s2.x: %d \n ", s2.x);
	printf("s2.y: %d \n ", s2.y);

	s.x = 1;
	s.y = 2;
	change(s);
	// s의 x,y 값은?
	printf("s.x: %d, s.y:%d \n", s.x, s.y);
}
