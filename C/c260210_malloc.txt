#include <stdio.h>
#include <stdlib.h>

struct Student {
    int id;
    int score;
};

int main() {
    // 1. 구조체 포인터 선언 및 메모리 할당
    struct Student* s1 = (struct Student*)malloc(sizeof(struct Student));

    if (s1 == NULL) return 1; // 할당 실패 시 예외 처리

    // 2. 멤버 접근 (화살표 연산자 사용)
    s1->id = 101;
    s1->score = 95;

    printf("학번: %d, 점수: %d\n", s1->id, s1->score);

    // 3. 메모리 해제
    free(s1);

    return 0;
}
