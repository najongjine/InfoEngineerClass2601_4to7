#include <stdio.h>
int test1() {
    return 1;
}
int main()
{
    /* c언어는 석기시대 언어라서 true, false가 기본으로 없어요
    0 == false
    !0 == true 
    if문은 if는 무조건 실행, else if 랑 else 는 옵션
    if() 안은 무조건 true, false로 치환됨
    */

    if(1) {
        int num1 = 1;
        printf("num1:%d \n",num1);
        if (1) {
            num1 = 2;
            if (1) {
                num1 = 3;
                printf("num1:%d \n", num1);
            }
        }
        printf("num1:%d \n", num1);
    }
    else {
        printf("false \n");
    }

    /* if() 안에 들어갈수 있는 정해져 있어요
    * 데이터
    * 비교연산
    * 논리연산
    * 산술연산
    * 함수
    */
    int num1 = 1;
    int num2 = 2;
    if (num1) {}
    if (num1 > num2+1-2) {}
    if (num1 <= num2) {}
    if (num1 >> 9) {}
    if(num1 && num2){}
    if (num1 || num2) {}
    if (num1 == num2) {}

    int arnum1[3] = { 1,3,5 };
    int arnum2[3] = { 1,3,5 };
    printf("arnum1: %p, arnum2: %p \n", arnum1, arnum2);
    //결과 보신 분들은, arnum1, arnum2 주소값 눈으로 확인해 보세요
    if (arnum1[1] >> 1  > arnum2[2]) {
        printf("arnum1[1] >> 1 크다 \n");
    }
    else {
        printf("arnum1[1] >> 1 작다 \n");
    }
    
}
