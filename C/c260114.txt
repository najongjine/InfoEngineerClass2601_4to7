// c260114.cpp : 이 파일에는 'main' 함수가 포함됩니다. 거기서 프로그램 실행이 시작되고 종료됩니다.
//

#include <stdio.h>
int test1() {
    return 1;
}
int main()
{
    /* c언어는 석기시대 언어라서 true, false가 기본으로 없어요
    0 == false
    !0 == true 
    if문은 if는 무조건 실행, else if 랑 else 는 옵션
    if() 안은 무조건 true, false로 치환됨
    */

    if(1) {
        int num1 = 1;
        printf("num1:%d \n",num1);
        if (1) {
            num1 = 2;
            if (1) {
                num1 = 3;
                printf("num1:%d \n", num1);
            }
        }
        printf("num1:%d \n", num1);
    }
    else {
        printf("false \n");
    }

    /* if() 안에 들어갈수 있는 정해져 있어요
    * 데이터
    * 비교연산
    * 논리연산
    * 산술연산
    * 함수
    */
    int num1 = 1;
    int num2 = 2;
    if (num1) {}
    if (num1 > num2+1-2) {}
    if (num1 <= num2) {}
    if (num1 >> 9) {}
    if(num1 && num2){}
    if (num1 || num2) {}
    if (num1 == num2) {}

    char arnum1[3] = { 'a','c','\0'};
    char arnum2[3] = { 'a','c','\0' };
    printf("arnum1: %p, arnum2: %p \n", arnum1, arnum2);
    //결과 보신 분들은, arnum1, arnum2 주소값 눈으로 확인해 보세요
    if (arnum1[0] == arnum2[0]) { // arnum2[2]
        printf("같다 \n");
    }
    else {
        printf("틀리다 \n");
    }

    /* 배열은 결국 포인터에요.
    포인터를 그냥 쌩으로 쓰면 어려워서 배열은 []로 원소에 접근해요  
    메모리 어느 공간에 연속된 공간을 만들고, 그곳의 시작 주소를
    변수에 담는다 */

    /* 90점 이상 A등급,
    80점 이상 B등급
    70점 이상 C등급
    60점 이상 D등급
    그 이하 F 등급 if문으로 판별기 만들어 보세요
    */
    long mygrade = 89;
    if (mygrade>=90) {
        printf("A등급 \n");
    }
    else if (mygrade>=80) {
        printf("B등급 \n");
    }
    else if (mygrade >= 70) {
        printf("C등급 \n");
    }
    else if (mygrade >= 60) {
        printf("D등급 \n");
    }
    else {
        printf("F등급 \n");
    }
    
    /* switch 는 if랑 비슷함
    근데, case 에는 상수밖에 못적음
    비교연산, 함수호출, 변수호출 그런거 못함
    단, break 조심 해야함.
    break 없으면 matching 되는 case 진입후,
    break 키워드가 없으면 그냥 쭉 실행함*/
    mygrade = 'A';
    switch (mygrade) {
        case 97:
            printf("97\n");
            break;
        case 65:
            printf("65\n");

        case 64:
            printf("64\n");

        case 63:
            printf("63\n");
            break;
        default:
            printf("몰라\n");
    }

    
    /* while 반복문
    * () 안의 조건이 true 여야지 반복 수행해요
    * 반복문을 빠져나오는 방법은 break를 만나거나
    * 조건 false가 되거나 둘중 하나
    */
    while (0) {
        printf("# 반복중");
        //break;
    }


}
