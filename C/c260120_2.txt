
#include <stdio.h>
struct Node {
	int n;
	struct Node* next;
};

int main()
{	
	struct Node a = { 10,NULL };
	struct Node b = { 20,NULL };
	struct Node c = { 30,NULL };

	struct Node* head = &a;
	a.next = &b;
	b.next = &c;

	int sum = 0;
	struct Node* p;
	for (p = head; p != NULL; p=p->next) {
		if (p->n > 15) sum += p->n;
	}
	printf("sum : %d \n", sum);
}

#include <stdio.h>
struct Node {
	int n;
	struct Node* link;
};

int main()
{	
	struct Node n1 = { 100,NULL };
	struct Node n2 = { 200,NULL };
	struct Node n3 = { 300,NULL };
	n1.link = &n2;
	n2.link = &n3;
	struct Node* p = &n1;
	if (p->link != NULL) {
		p->link = p->link->link;
	}
	int total = 0;
	for (; p != NULL;p=p->link) {
		total += p->n;
	}
	printf("%d \n", total);
}
