
#include <stdio.h>
struct Node {
	int n;
	struct Node* next;
};

int main()
{	
	struct Node a = { 10,NULL };
	struct Node b = { 20,NULL };
	struct Node c = { 30,NULL };

	struct Node* head = &a;
	a.next = &b;
	b.next = &c;

	int sum = 0;
	struct Node* p;
	for (p = head; p != NULL; p=p->next) {
		if (p->n > 15) sum += p->n;
	}
	printf("sum : %d \n", sum);
}
