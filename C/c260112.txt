#include "stdio.h"

/*
* 한 scope 안에는 동일 이름의 변수가 또 존재할수 없어요
* ex: int a; int a; 이건 에러
*/
int main()
{
    /*
    * int := 메모리에 4byte 공간의 크기를 잡음
    * a := 메모리 어딘가에 a 라는 별명 붙임(int 에서 크기 잡은거)
    * = := 대입 연산자.  변수 = data, 변수 공간에 data를 복붙 하겠다
    * a=-2  이미 자리잡은 a 공간에 -2 복붙 하겠다. 예전 데이터 사라짐
    */
    int a= 1;
    a = -2;
    
    /* b 라는 int 형 변수를 선언 하고, 2 라는 값으로 초기화 해주세요 */
    int b = 2;

    /* b 라는 공간에, b의 공간에서 데이터 복붙해서 가져온후,
    1을 더하고, 그 결과값을 b 에다가 저장하기 */
    b = b+1;

    int c;
    c = a + b;
    c = c + 1;
    c += 1;
    printf("%d \n", c);
}
