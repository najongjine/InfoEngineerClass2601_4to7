// c260112.cpp : 이 파일에는 'main' 함수가 포함됩니다. 거기서 프로그램 실행이 시작되고 종료됩니다.

#include "stdio.h"

/*
* 한 scope 안에는 동일 이름의 변수가 또 존재할수 없어요
* ex: int a; int a; 이건 에러
*/
int main()
{
    /*
    * int := 메모리에 4byte 공간의 크기를 잡음
    * a := 메모리 어딘가에 a 라는 별명 붙임(int 에서 크기 잡은거)
    * = := 대입 연산자.  변수 = data, 변수 공간에 data를 복붙 하겠다
    * a=-2  이미 자리잡은 a 공간에 -2 복붙 하겠다. 예전 데이터 사라짐
    */
    int a= 1;
    a = -2;
    
    /* b 라는 int 형 변수를 선언 하고, 2 라는 값으로 초기화 해주세요 */
    int b = 2;

    /* b 라는 공간에, b의 공간에서 데이터 복붙해서 가져온후,
    1을 더하고, 그 결과값을 b 에다가 저장하기 */
    b = b+1;

    int c;
    c = a + b;
    c = c + 1;
    c += 1;
    

    /*
    * 컴퓨터는 계산기에요
    * 근데 문방구에 파는 계산기와 다른점은,
    * + / * - %
    * if() for()
    * 함수()
    * 요런게 추가가 됬어요
    * 
    * 우리는 폰 노이만 아키텍처를 쓰고있고,
    * 폰 노이만 모델의 가장 중요한점은 메모리에요
    */
    b = 1;
    a = 2;
    
    float f1 = 1;
    /* %f 는 소수 자리수 정하기 기능이 있어요
    ex: %.2f */
    /* 옛날언어(c,java) 얘네들은 나누기 할때 조심 해야되요
    int / int = 소수점 버리기에요 
    (int)1/(int)2 = 0*/
    f1 = b / a;
    printf("f1: %.2f \n", f1);

    f1 = (float)b / 2;
    printf("(float)b / 2: %.2f \n", f1);
}
