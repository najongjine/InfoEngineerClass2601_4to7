
#include <stdio.h>


int main()
{
	/* 데이터 & 데이터 := 비트 연산
	데이터 && 데이터 := 논리 연산
	&데이터 := 주소 뽑아내기 */
	int a = 1;
	printf("a 의 주소: %p \n", &a);
	printf("a 가 가진 값: %d \n", a);

	int * pa = &a;
	printf("pa 의 주소: %p \n", &pa);
	printf("pa 가 가진 값: %p \n", pa);
	printf("pa 주소의 원본값: %d \n", *pa);
	printf("pa 주소의 원본값: %d \n", pa[0]);

}
