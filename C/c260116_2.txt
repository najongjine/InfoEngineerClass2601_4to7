
#include <stdio.h>
// { 10,55,23,7,88,30 };
int* get_min_max(int* arr, int size, int* max, int* min) {
	printf(" --------------- \n");
	printf("arr 주소: %d \n", arr);
	printf("max 주소: %d \n", max);
	printf("min 주소: %d \n", min);
	*max = *arr;
	*min = *arr;
	printf("%d 원산지에 값 넣기: %d \n", max, *arr);
	printf("%d 원산지에 값 넣기: %d \n", min, *arr);
	printf("---------\n");

	for (int i = 1; i < size;i++) {
		printf("*(%d + %d) > %d \n", arr, i, *max);
		if (*(arr + i) > *max) {
			*max = *(arr + i);
			printf("*(%d)= *(%d + %d) \n", max, arr, i);
		}
		printf("*(%d + %d) < %d \n", arr, i, *max);
		if (*(arr+i) < *min) {
			*min = *(arr + i);
			printf("*(%d)= *(%d + %d) \n", min, arr, i);
		}

	}
}


int main()
{
	/* 데이터 & 데이터 := 비트 연산
	데이터 && 데이터 := 논리 연산
	&데이터 := 주소 뽑아내기 */
	int a = 1;
	printf("a 의 주소: %p \n", &a);
	printf("a 가 가진 값: %d \n", a);

	int * pa = &a;
	printf("pa 의 주소: %p \n", &pa);
	printf("pa 가 가진 값: %p \n", pa);
	printf("pa 주소의 원본값: %d \n", *pa);
	printf("pa 주소의 원본값: %d \n", pa[0]);

	a = 10;
	pa = &a;
	*pa = 20;
	printf("a: %d \n", a);

	int nums[] = { 10,55,23,7,88,30 };
	int size = sizeof(nums) / sizeof(nums[0]);
	int max_val, min_val;
	get_min_max(nums, size, &max_val, &min_val);
	printf("max: %d \n", max_val);
	printf("min: %d \n", min_val);

}
