void reverseStr(char* str) {
	int len = 0;
	while (str[len] != '\0') {
		len++;
	}
	char temp;
	for (int i = 0; i < len/2;i++) {
		temp = str[i];
		*(str + i) = str[len - 1 - i];
		str[len - 1 - i] = temp;
	}
}
