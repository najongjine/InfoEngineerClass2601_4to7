#include <stdio.h>

int main()
{
    int int1 = 1;
    int int2 = 2;
    float f1 = 1.0;

    /* int 수학수식 int = int */
    //printf("%d / %d : %d \n",int1,int2,int1/int2);

    /* float 수학수식 int = int | float */
    //printf("%f / %d : %f \n", f1, int2, f1 / int2);

    int1 = 0;
    /* ++변수 := 전위연산(가장먼저). 가장 사람입장에서 상식적임
    뭐가 있던간에, 변수 찾아가서 있는거 값 증가 시켜라 */
    printf("int1: %d \n", ++int1);
    printf("int1: %d \n", int1);

    int1 = 0;
    /* 변수++ := 후위연산(재일 나중). 가장 사람입장에서 비상식점
    뭐가 있던간에, 맨 나중에 변수 찾아가서 있는거 값 증가 시켜라 */
    printf("int1: %d \n", int1++);
    printf("int1: %d \n", int1);

    int1 = 1;
    int2 = 2;
    int int3 = int1++ + int2++;
    printf("int3: %d, int1:%d, int2:%d \n", int3, int1, int2);

    char ch1 = 'a'; // 상수 테이블 id 97
    char ch2 = 'A'; // 상수 테이블 id 65
    char ch0 = '0'; // 상수 테이블 id 48
    printf("\n\n %c \n", ch1-32);

    putchar(ch1); // 문자 하나 출력하고 다음줄 개행
    puts("dfddf");// 문자열 보여주고 다음줄 개행

    int1 = 1;
    int2 = 2;
    printf("!int1 :%d \n", !int2); // ! := 반전
    // sizeof:= 데이터타입 크기 알아내기
    printf("sizeof(int1) : %d \n", sizeof(int1));

    /* 비트 산술연산자보다 우선순위가 낮아요
    * {변수 비트연산 변수} 나오면 이진수 연산
    * & | ^ << >> 
    * (& && | || *) := c 언어에서 제일 어려운놈. 왜냐면 문맥에 따라서 
    * 연산하는 방법이 틀려요
    */
    printf("\n\n\n");
    int1 = 1;
    int2 = 2;
    printf("int1 & int2 : %d \n", int1 & int2); // AND
    printf("int1 | int2 : %d \n", int1 | int2); // OR
    printf("int1 ^ int2 : %d \n", int1 ^ int2); // XOR
    printf("int2 << 1 : %d \n", int2 << 1); // 4
    printf("int2 >> 1 : %d \n", int2 >> 1); // 1

    printf("\n\n\n");
    int1 = 1;
    int2 = 2;
    int3 = 1 + int1 << 2;
    printf("int3(1 + int1 << 2)=%d \n", int3); // 1+int1 부터 먼저 함
    int3 = int1 >> 1 + int2 | int1;
    printf("int1 >> 1 + int2 | int1 = %d \n", int3); // 1+int2 먼저 함

    /* 기본 자료형:short, int, long , float, char */

    int arint1[3] = { 1,3,5 };
    char arch1[3] = "ab";
    char arch2[3] = {'a','b','\0'};
    printf("arch1 : %s \n",arch1); // "ab"
    printf("arch2 : %s \n", arch2); // "ab"
}
