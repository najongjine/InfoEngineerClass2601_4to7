
#include <stdio.h>
/* stamp 결합도. 원본을(주소)를 주는것
원본 내용이 바뀜 */
void shuffleArray(int ar1[], int length) {
    int temp = ar1[0];
    ar1[0] = ar1[1];
    ar1[1] = temp;
}
void shuffleArray2(char car1[]) {
    char temp = car1[0];
    car1[0] = car1[1];
    car1[1] = temp;
}
void printArray(int ar1[],int length) {
    for (int i = 0; i < length; i++) {
        printf("%d ", ar1[i]);
    }
    printf("\n");
}

int main()
{
    /* ar1 은 {} 배열의 시작 주소를 저장하고 있어요 */
    int ar1[3] = { 1,2,3 };
    printf("ar1: %p \n",ar1);
    printf("ar1[0]: %d \n",ar1[0]);
    printf("*(ar1+0): %d \n", *(ar1+0));
    /* shuffleArray의 매게변수에 ar1이 가지고 있는 주소
    ({}배열의 시작 주소) 를 복붙해서 주는 거에요 
    이걸 stamp 결합도라고 해요 */
    shuffleArray(ar1, sizeof(ar1) / sizeof(ar1[0]));
    printArray(ar1,sizeof(ar1)/sizeof(ar1[0]));

    /* c 언어는 문자열을 포인터로 관리해요
    배열도 포인터고, 타입 *  이것도 포인터에요 */
    char car1[5] = "abc";
    char* car2 = car1;
    car2[0] = car2[1];
    

    car1[5] = "abc";

    shuffleArray2(car1);
    printf("car1: %s \n", car1);

}
