package com.java_4to7;

class Animal {
	int leg=4;
	void makeSound() {
		System.out.println("으러렁");
	}
}
/* 어떤 클래스를 extends 로 받으면 이걸 상속 받았다 하고
 * Animal(부모) <- Dog(자식) 이렇게 도형으로 그릴수도 있다
 * 상속받은 자식은 부모의 모든것을 다 가져갈수 있다 
 * 
 * 헌데... 부모랑 자식에 똑같은 이름의 변수가 있다?
 * 그러면 그냥 자기꺼 변수를 쓴다. 절대 부모변수 값 + 자식값 아니다
 * */
class Dog extends Animal{
	boolean fur=true;
	int leg=8;
}

public class Java260121 {

	public static void main(String[] args) {
		Animal animal=new Animal();
		animal.makeSound();
		Dog dog=new Dog();
		System.out.println(dog.leg);
		
	}

}

package com.java_4to7;

class Animal {
	/* 클래스는 만들어질때 항상 생성자가 발동한다 
	 * 만약, 생성자를 하나도 안만들었으면, 지가 껍데기 생성자를
	 * 컴파일 단계에서 만든다 */
	
	int leg=4;
	Animal(int _leg){
		leg=_leg;
	}
	void makeSound() {
		System.out.println("으러렁");
	}
}



public class Java260121 {

	public static void main(String[] args) {
		Animal animal=new Animal();
		animal.makeSound();
		
	}

}

package com.java_4to7;

class Animal {
	/* 클래스는 만들어질때 항상 생성자가 발동한다 
	 * 만약, 생성자를 하나도 안만들었으면, 지가 껍데기 생성자를
	 * 컴파일 단계에서 만든다 */
	
	int leg=4;
	Animal(){
		System.out.println("animal 탄생");
	}
	Animal(int _leg){
		this();
		leg=_leg;
	}

	void makeSound() {
		System.out.println("으러렁");
	}
}



public class Java260121 {

	public static void main(String[] args) {
		Animal animal=new Animal(2);
		animal.makeSound();
		
	}

}
