package com.java_4to7;
class Engine{
	String type;
	public Engine(String type) {
		this.type=type;
	}
	public void start() {
		System.out.println(type+"엔진 가동");
	}
}
class Car{
	String model;
	Engine engine; // Has-A
	Car(String model, Engine engine){
		this.model=model;
		this.engine=engine;
	}
	public void drive() {
		System.out.println(model+"주행을 시작");
		engine.start();
	}
}
public class Java260121 {
	public static void main(String[] args) {
		Engine myEngin=new Engine("v8");
		Car mycar=new Car("제네시스",myEngin);
		mycar.drive();
		
	}

}

package com.java_4to7;
class Processor{
	String model;
	Processor(String m){
		this.model= m;
		System.out.println("P-Init");
	}
	void process() {
		System.out.println(model + "동작");
	}
}
class SmartPhone{
	String name;
	Processor cpu;
	
	SmartPhone(String name, String cpuModel){
		this.name=name;
		this.cpu=new Processor(cpuModel);
		System.out.println("S-Init");
	}
	void run() {
		System.out.print(name+":");
		cpu.process();
	}
}
public class Java260121 {
	public static void main(String[] args) {
		SmartPhone sp=new SmartPhone("갤럭시", "AMD");
		sp.run();
		
	}

}
